6   package gov.usgs.volcanoes.wwsclient;
7   
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  import com.martiansoftware.jsap.FlaggedOption;
12  import com.martiansoftware.jsap.JSAP;
13  import com.martiansoftware.jsap.JSAPResult;
14  import com.martiansoftware.jsap.Parameter;
15  import com.martiansoftware.jsap.Switch;
16  
17  import gov.usgs.volcanoes.core.args.Args;
18  import gov.usgs.volcanoes.core.args.ArgumentException;
19  import gov.usgs.volcanoes.core.args.Arguments;
20  import gov.usgs.volcanoes.core.args.decorator.ScnlArg;
21  import gov.usgs.volcanoes.core.args.decorator.TimeSpanArg;
22  import gov.usgs.volcanoes.core.args.decorator.VerboseArg;
23  import gov.usgs.volcanoes.core.data.Scnl;
24  import gov.usgs.volcanoes.core.time.TimeSpan;
25  
26  /**
27   * Argument processor for WWSClient.
28   *
29   * @author Tom Parker
30   */
31  public class WwsClientArgs {
32    private static final Logger LOGGER = LoggerFactory.getLogger(WwsClientArgs.class);
33  
34    /** format of time on cmd line */
35    public static final String INPUT_TIME_FORMAT = "yyyyMMddHHmm";
36  
37    private static final String PROGRAM_NAME = "java -jar gov.usgs.volcanoes.winston.WWSClient";
38    private static final String EXPLANATION = "I am the Winston Wave Server client.\n";
39  
40    private static final Parameter[] PARAMETERS = new Parameter[] {
41        new FlaggedOption("server", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.REQUIRED, 's', "server",
42            "Remote server."),
43        new FlaggedOption("port", JSAP.INTEGER_PARSER, "16022", JSAP.NOT_REQUIRED, 'p', "port",
44            "Remote port."),
45        new FlaggedOption("command", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.NOT_REQUIRED,
46            JSAP.NO_SHORTFLAG, "command", "Command string."),
47        new Switch("sac", JSAP.NO_SHORTFLAG, "sac", "Write SAC output."),
48        new Switch("txt", JSAP.NO_SHORTFLAG, "txt", "Write samples as text output."),
49        new Switch("rsam", JSAP.NO_SHORTFLAG, "rsam", "Write RSAM output."),
50        new FlaggedOption("rsamPeriod", JSAP.INTEGER_PARSER, "300", JSAP.NOT_REQUIRED,
51            JSAP.NO_SHORTFLAG, "rsamPeriod", "RSAM period."),
52  
53        new Switch("heli", JSAP.NO_SHORTFLAG, "heli", "Write Helicorder output."),
54        new Switch("menu", JSAP.NO_SHORTFLAG, "menu", "Retrieve server menu.")};
55  
56    /** If true, log more. */
57    public final boolean verbose;
58  
59    /** Remote server. */
60    public final String server;
61  
62    /** Remote server port. */
63    public final int port;
64  
65    /** Time span to retrieve. */
66    public final TimeSpan timeSpan;
67  
68    /** channel to query */
69    public final Scnl channel;
70  
71    /** If true, print server menu */
72    public final boolean menu;
73  
74    /** If true, write SAC output. */
75    public final boolean sacOutput;
76  
77    /** If true, write TXT output. */
78    public final boolean txtOutput;
79  
80    /** If true, write RSAM output. */
81    public final boolean rsamOutput;
82  
83    /** Send provided command directly to wave server and return result to <STDOUT> */
84    public final String command;
85  
86    /** RSAM period in seconds */
87    public final int rsamPeriod;
88  
89    /** If true, write Heli output. */
90    public final boolean heliOutput;
91  
92    /**
93     * Class constructor.
94     * 
95     * @param commandLineArgs the command line arguments
96     * @throws ArgumentException  when things go wrong
97     */
98    public WwsClientArgs(final String[] commandLineArgs) throws ArgumentException  {
99      Arguments args = null;
100     args = new Args(PROGRAM_NAME, EXPLANATION, PARAMETERS);
101     args = new ScnlArg(false, args);
102     args = new TimeSpanArg(INPUT_TIME_FORMAT, false, args);
103     args = new VerboseArg(args);
104 
105     JSAPResult jsapResult = null;
106     jsapResult = args.parse(commandLineArgs);
107 
108     if (!jsapResult.success() && !jsapResult.getBoolean("help")) {
109 
110       throw new ArgumentException("I'm not sure what to do. Try --help");
111       }
112 
113     verbose = jsapResult.getBoolean("verbose");
114     timeSpan = (TimeSpan) jsapResult.getObject("timeSpan");
115     server = jsapResult.getString("server");
116     port = jsapResult.getInt("port");
117     channel = (Scnl) jsapResult.getObject("channel");
118     menu = jsapResult.getBoolean("menu");
119     sacOutput = jsapResult.getBoolean("sac");
120     txtOutput = jsapResult.getBoolean("txt");
121     command = jsapResult.getString("command");
122     rsamOutput = jsapResult.getBoolean("rsam");
123     rsamPeriod = jsapResult.getInt("rsamPeriod");
124     heliOutput = jsapResult.getBoolean("heli");
125 
126     if (!jsapResult.getBoolean("help")) {
127       LOGGER.debug("Setting: verbose={}", verbose);
128       LOGGER.debug("Setting: timeSpan={}", timeSpan);
129       LOGGER.debug("Setting: server={}", server);
130       LOGGER.debug("Setting: port={}", port);
131       LOGGER.debug("Setting menu={}", menu);
132       LOGGER.debug("Setting sacOutput={}", sacOutput);
133       LOGGER.debug("Setting txtOutput={}", txtOutput);
134       LOGGER.debug("Setting rsamOutput={}", rsamOutput);
135       LOGGER.debug("Setting heliOutput={}", heliOutput);
136     }
137 
138     if (sacOutput || txtOutput || rsamOutput || heliOutput) {
139       if (channel == null) {
140         throw new ArgumentException("No channel provided.");
141       }
142       if (timeSpan == null) {
143         throw new ArgumentException("No time span provided.");
144       }
145     }
146 
147   }
