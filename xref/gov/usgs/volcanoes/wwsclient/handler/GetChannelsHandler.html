21  /**
22   * Receive and process response from a winston GETMENU request.
23   *
24   * @author Tom Parker
25   */
26  public class GetChannelsHandler extends AbstractCommandHandler {
27    private static final Logger LOGGER = LoggerFactory.getLogger(GetChannelsHandler.class);
28    private static final String DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
29  
30    private int linesTotal;
31    private int linesRead;
32    private final List<gov.usgs.volcanoes.winston.Channel> channels;
33    private StringBuffer menu;
34  
35    /**
36     * Constructor
37     * 
38     * @param channels List to be populated with channels
39     */
40    public GetChannelsHandler(List<gov.usgs.volcanoes.winston.Channel> channels) {
41      super();
42      linesTotal = -Integer.MAX_VALUE;
43      linesRead = 0;
44      menu = new StringBuffer();
45      this.channels = channels;
46    }
47  
48    @Override
49    public void handle(ByteBuf msgBuf) throws IOException {
50      if (linesTotal < 0) {
51        String header = ClientUtils.readResponseHeader(msgBuf);
52        if (header == null) {
53          LOGGER.debug("Still waiting for full response line.");
54          return;
55        } else {
56          linesTotal = Integer.parseInt(header.split(" ")[1]);
57          LOGGER.debug("Server has {} channels.", linesTotal);
58        }
59      }
60  
61      byte[] bytes = new byte[msgBuf.readableBytes()];
62      msgBuf.readBytes(bytes);
63      String chunk = new String(bytes);
64      linesRead += countLines(chunk);
65      menu.append(chunk);
66      if (linesRead == linesTotal) {
67        for (String line : menu.toString().split("\n")) {
68          try {
69            channels.add(new Channel.Builder().parse(line).build());
70          } catch (UtilException e) {
71            LOGGER.error("Unable to parse server response: {}", line);
72          }
73        }
74        sem.release();
75      } else {
76        LOGGER.debug("Read {} of {} channels", linesRead, linesTotal);
77      }
78    }
79  
80  
81    private int countLines(String buf) {
82      int lines = 0;
83      for (int pos = 0; pos < buf.length(); pos++) {
84        if (buf.charAt(pos) == '\n') {
85          lines++;
86        }
87      }
88      return lines;
89    }
90  }
